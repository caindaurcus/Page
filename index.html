Auto Potion
Usa poción automáticamente cuando la vida baja del 30%
-- Auto Potion Script
local HP_THRESHOLD = 0.30

function onUpdate()
  local hp = Player.getHP() / Player.getMaxHP()
  if hp < HP_THRESHOLD then
    Item.use("Health Potion")
  end
end
Auto Loot Filter
Filtra loot automáticamente por rareza 4 o superior
-- Loot Filter: rango 4+
local MIN_RARITY = 4

function onItemDrop(item)
  if item.rarity >= MIN_RARITY then
    Player.pickup(item)
    Log.print("Recogido: " .. item.name)
  end
end
Skill Combo — Assassin
Combo PvP: Shadow Blink → Void Slash x3 → Death Mark
-- Assassin PvP Combo
function pvpCombo(target)
  Skill.cast("Shadow Blink", target)
  wait(200)
  for i = 1, 3 do
    Skill.cast("Void Slash", target)
    wait(150)
  end
  Skill.cast("Death Mark", target)
end
